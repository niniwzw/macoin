64 std::string static EncodeDumpString(const std::string &str) {
65  std::stringstream ret;
66  BOOST_FOREACH(unsigned char c, str) {
67  if (c <= 32 || c >= 128 || c == '%') {
68  ret << '%' << HexStr(&c, &c + 1);

70  ret << c;
71  }
72  }

76 std::string DecodeDumpString(const std::string &str) {
77  std::stringstream ret;
78  for (unsigned int pos = 0; pos < str.length(); pos++) {
79  unsigned char c = str[pos];
80  if (c == '%' && pos+2 < str.length()) {
81  c = (((str[pos+1]>>6)*9+((str[pos+1]-'0')&15)) << 4) |
82  ((str[pos+2]>>6)*9+((str[pos+2]-'0')&15));
83  pos += 2;
84  }
85  ret << c;
86  }
87  return ret.str();
88 }
