92  for (QList<QPair<QString, QString> >::iterator i = items.begin(); i != items.end(); i++)
93  {
94  bool fShouldReturnFalse = false;
95  if (i->first.startsWith("req-"))

97  i->first.remove(0, 4);
98  fShouldReturnFalse = true;
99  }
100 
101  if (i->first == "label")
102  {
103  rv.label = i->second;
104  fShouldReturnFalse = false;
105  }
106  else if (i->first == "amount")
107  {
108  if(!i->second.isEmpty())
109  {

111  {
112  return false;
113  }
114  }
115  fShouldReturnFalse = false;
116  }
117 
118  if (fShouldReturnFalse)
119  return false;
120  }
121  if(out)
122  {
123  *out = rv;
124  }
125  return true;
126 }
