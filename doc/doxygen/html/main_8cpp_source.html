414 bool IsStandardTx(const CTransaction& tx)
415 {
416  if (tx.nVersion > CTransaction::CURRENT_VERSION)
417  return false;
418 
419  // Treat non-final transactions as non-standard to prevent a specific type
420  // of double-spend attack, as well as DoS attacks. (if the transaction
421  // can't be mined, the attacker isn't expending resources broadcasting it)
422  // Basically we don't want to propagate transactions that can't included in
423  // the next block.
424  //
425  // However, IsFinalTx() is confusing... Without arguments, it uses
426  // chainActive.Height() to evaluate nLockTime; when a block is accepted, chainActive.Height()
427  // is set to the value of nHeight in the block. However, when IsFinalTx()
428  // is called within CBlock::AcceptBlock(), the height of the block *being*
429  // evaluated is what is used. Thus if we want to know if a transaction can
430  // be part of the *next* block, we need to call IsFinalTx() with one more
431  // than chainActive.Height().
432  //
433  // Timestamps on the other hand don't get any special treatment, because we
434  // can't know what timestamp the next block will have, and there aren't
435  // timestamp applications where it matters.
436  if (!IsFinalTx(tx, nBestHeight + 1)) {
437  return false;
438  }
439  // nTime has different purpose from nLockTime but can be used in similar attacks
440  if (tx.nTime > FutureDrift(GetAdjustedTime(), nBestHeight + 1)) {
441  return false;
442  }
443 
444  // Extremely large transactions with lots of inputs can cost the network
445  // almost as much to process as they cost the sender in fees, because
446  // computing signature hashes is O(ninputs*txsize). Limiting transactions
447  // to MAX_STANDARD_TX_SIZE mitigates CPU exhaustion attacks.
448  unsigned int sz = tx.GetSerializeSize(SER_NETWORK, CTransaction::CURRENT_VERSION);
449  if (sz >= MAX_STANDARD_TX_SIZE)
450  return false;
451 
452  BOOST_FOREACH(const CTxIn& txin, tx.vin)
453  {
454  // Biggest 'standard' txin is a 3-signature 3-of-3 CHECKMULTISIG
455  // pay-to-script-hash, which is 3 ~80-byte signatures, 3
456  // ~65-byte public keys, plus a few script ops.
457  if (txin.scriptSig.size() > 500)
458  return false;
459  if (!txin.scriptSig.IsPushOnly())
460  return false;
461  if (!txin.scriptSig.HasCanonicalPushes()) {
462  return false;
463  }
464  }
465 
466  unsigned int nDataOut = 0;
467  txnouttype whichType;
468  BOOST_FOREACH(const CTxOut& txout, tx.vout) {
469  if (!::IsStandard(txout.scriptPubKey, whichType))
470  return false;
471  if (whichType == TX_NULL_DATA)
472  nDataOut++;
473  if (txout.nValue == 0)
474  return false;
475  if (!txout.scriptPubKey.HasCanonicalPushes()) {
476  return false;
477  }
478  }
479 
480  // only one OP_RETURN txout is permitted
481  if (nDataOut > 1) {
482  return false;
483  }
484 
485  return true;
486 }

506 //
507 // Check transaction inputs, and make sure any
508 // pay-to-script-hash transactions are evaluating IsStandard scripts
509 //
510 // Why bother? To avoid denial-of-service attacks; an attacker
511 // can submit a standard HASH... OP_EQUAL transaction,
512 // which will get accepted into blocks. The redemption
513 // script can be anything; an attacker could use a very
514 // expensive-to-check-upon-redemption script like:
515 // DUP CHECKSIG DROP ... repeated 100 times... OP_1
516 //
517 bool CTransaction::AreInputsStandard(const MapPrevTx& mapInputs) const
518 {
519  if (IsCoinBase() || IsBack())
520  return true; // Coinbases don't use vin normally
521 
522  for (unsigned int i = 0; i < vin.size(); i++)
523  {
524  const CTxOut& prev = GetOutputFor(vin[i], mapInputs);
525 
526  vector<vector<unsigned char> > vSolutions;
527  txnouttype whichType;
528  // get the scriptPubKey corresponding to this input:
529  const CScript& prevScript = prev.scriptPubKey;
530  if (!Solver(prevScript, whichType, vSolutions))
531  return false;
532  int nArgsExpected = ScriptSigArgsExpected(whichType, vSolutions);
533  if (nArgsExpected < 0)
534  return false;
535 
536  // Transactions with extra stuff in their scriptSigs are
537  // non-standard. Note that this EvalScript() call will
538  // be quick, because if there are any operations
539  // beside "push data" in the scriptSig the
540  // IsStandard() call returns false
541  vector<vector<unsigned char> > stack;
542  if (!EvalScript(stack, vin[i].scriptSig, *this, i, 0))
543  return false;
544 
545  if (whichType == TX_SCRIPTHASH)
546  {
547  if (stack.empty())
548  return false;
549  CScript subscript(stack.back().begin(), stack.back().end());
550  vector<vector<unsigned char> > vSolutions2;
551  txnouttype whichType2;
552  if (!Solver(subscript, whichType2, vSolutions2))
553  return false;
554  if (whichType2 == TX_SCRIPTHASH)
555  return false;
556 
557  int tmpExpected;
558  tmpExpected = ScriptSigArgsExpected(whichType2, vSolutions2);
559  if (tmpExpected < 0)
560  return false;
561  nArgsExpected += tmpExpected;
562  }
563 
564  if (stack.size() != (unsigned int)nArgsExpected)
565  return false;
566  }
567 
568  return true;
569 }

692 {

705  }

710  if (nNewBlockSize >= MAX_BLOCK_SIZE_GEN)
711  return MAX_MONEY;
712  nMinFee *= MAX_BLOCK_SIZE_GEN / (MAX_BLOCK_SIZE_GEN - nNewBlockSize);
713  }
714 
715  if (!MoneyRange(nMinFee))
716  nMinFee = MAX_MONEY;
717  return nMinFee;

720 bool AcceptToMemoryPool(CTxMemPool& pool, CTransaction &tx,
721  bool* pfMissingInputs)
722 {
723  AssertLockHeld(cs_main);
724  if (pfMissingInputs)
725  *pfMissingInputs = false;
726 
727  if (!tx.CheckTransaction())
728  return error("AcceptToMemoryPool : CheckTransaction failed");
729 
730  // Coinbase is only valid in a block, not as a loose transaction
731  if (tx.IsCoinBase())
732  return tx.DoS(100, error("AcceptToMemoryPool : coinbase as individual tx"));
733 
734  // ppcoin: coinstake is also only valid in a block, not as a loose transaction
735  if (tx.IsCoinStake())
736  return tx.DoS(100, error("AcceptToMemoryPool : coinstake as individual tx"));
737 
738  // Rather not work on nonstandard transactions (unless -testnet)
739  if (!fTestNet && !IsStandardTx(tx))
740  return error("AcceptToMemoryPool : nonstandard transaction type");
741 
742  // is it already in the memory pool?
743  uint256 hash = tx.GetHash();
744  if (pool.exists(hash)) {
745  return error("pool.exists(hash)");
746  }
747 
748  // Check for conflicts with in-memory transactions
749  {
750  LOCK(pool.cs); // protect pool.mapNextTx
751  for (unsigned int i = 0; i < tx.vin.size(); i++)
752  {
753  COutPoint outpoint = tx.vin[i].prevout;
754  if (pool.mapNextTx.count(outpoint))
755  {
756  // Disable replacement feature for now
757  return error("Check for conflicts with in-memory transactions");
758  }
759  }
760  }
761 
762  {
763  CTxDB txdb("r");
764 
765  // do we already have it?
766  if (txdb.ContainsTx(hash)) {
767  return error("txdb.ContainsTx(hash)");
768  }
769  if (!tx.IsBack()) {
770  MapPrevTx mapInputs;
771  map<uint256, CTxIndex> mapUnused;
772  bool fInvalid = false;
773  if (!tx.FetchInputs(txdb, mapUnused, false, false, mapInputs, fInvalid))
774  {
775  if (fInvalid)
776  return error("AcceptToMemoryPool : FetchInputs found invalid tx %s", hash.ToString().substr(0,10).c_str());
777  if (pfMissingInputs)
778  *pfMissingInputs = true;
779  return false;
780  }
781 
782  // Check for non-standard pay-to-script-hash in inputs
783  if (!tx.AreInputsStandard(mapInputs) && !fTestNet)
784  return error("AcceptToMemoryPool : nonstandard transaction input");
785 
786  // Note: if you modify this code to accept non-standard transactions, then
787  // you should add code here to check that the transaction does a
788  // reasonable number of ECDSA signature verifications.
789 
790  int64_t nFees = tx.GetValueIn(mapInputs)-tx.GetValueOut();
791  unsigned int nSize = ::GetSerializeSize(tx, SER_NETWORK, PROTOCOL_VERSION);
792 
793  // Don't accept it if it can't get into a block
794  int64_t txMinFee = tx.GetMinFee(1000, GMF_RELAY, nSize);
795  if (nFees < txMinFee)
796  return error("AcceptToMemoryPool : not enough fees %s, %"PRId64" < %"PRId64,
797  hash.ToString().c_str(),
798  nFees, txMinFee);
799 
800  // Continuously rate-limit free transactions
801  // This mitigates 'penny-flooding' -- sending thousands of free transactions just to
802  // be annoying or make others' transactions take longer to confirm.
803  if (nFees < MIN_RELAY_TX_FEE)
804  {
805  static CCriticalSection cs;
806  static double dFreeCount;
807  static int64_t nLastTime;
808  int64_t nNow = GetTime();
809 
810  {
811  LOCK(pool.cs);
812  // Use an exponentially decaying ~10-minute window:
813  dFreeCount *= pow(1.0 - 1.0/600.0, (double)(nNow - nLastTime));
814  nLastTime = nNow;
815  // -limitfreerelay unit is thousand-bytes-per-minute
816  // At default rate it would take over a month to fill 1GB
817  if (dFreeCount > GetArg("-limitfreerelay", 15)*10*1000 && !IsFromMe(tx))
818  return error("AcceptToMemoryPool : free transaction rejected by rate limiter");
819  if (fDebug)
820  printf("Rate limit dFreeCount: %g => %g\n", dFreeCount, dFreeCount+nSize);
821  dFreeCount += nSize;
822  }
823  }
824 
825  // Check against previous transactions
826  // This is done last to help prevent CPU exhaustion denial-of-service attacks.
827  if (!tx.ConnectInputs(txdb, mapInputs, mapUnused, CDiskTxPos(1,1,1), pindexBest, false, false))
828  {
829  return error("AcceptToMemoryPool : ConnectInputs failed %s", hash.ToString().substr(0,10).c_str());
830  }
831  }
832  }
833  pool.addUnchecked(hash, tx);
834  // Store transaction in memory
835  //if (!tx.IsBack())
836  //{
837  SyncWithWallets(tx, NULL, true);
838  //}
839  printf("AcceptToMemoryPool : accepted %s (poolsz %"PRIszu")\n",
840  hash.ToString().substr(0,10).c_str(),
841  pool.mapTx.size());
842  return true;
843 }
