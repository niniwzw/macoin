519 void BitcoinGUI::setNumBlocks(int count, int nTotalBlocks)
520 {
521  // don't show / hide progress bar and its label if we have no connection to the network
522  if (!clientModel || clientModel->getNumConnections() == 0)
523  {
524  progressBarLabel->setVisible(false);
525  progressBar->setVisible(false);
526 
527  return;
528  }
529 
530  QString strStatusBarWarnings = clientModel->getStatusBarWarnings();
531  QString tooltip;
532 
533  if(count < nTotalBlocks)
534  {
535  int nRemainingBlocks = nTotalBlocks - count;
536  float nPercentageDone = count / (nTotalBlocks * 0.01f);
537 
538  if (strStatusBarWarnings.isEmpty())
539  {
540  progressBarLabel->setText(tr("Synchronizing with network..."));
541  progressBarLabel->setVisible(true);
542  progressBar->setFormat(tr("~%n block(s) remaining", "", nRemainingBlocks));
543  progressBar->setMaximum(nTotalBlocks);
544  progressBar->setValue(count);
545  progressBar->setVisible(true);
546  }
547 
548  tooltip = tr("Downloaded %1 of %2 blocks of transaction history (%3% done).").arg(count).arg(nTotalBlocks).arg(nPercentageDone, 0, 'f', 2);
549  }
550  else
551  {
552  if (strStatusBarWarnings.isEmpty())
553  progressBarLabel->setVisible(false);
554 
555  progressBar->setVisible(false);
556  tooltip = tr("Downloaded %1 blocks of transaction history.").arg(count);
557  }
558 
559  // Override progressBarLabel text and hide progress bar, when we have warnings to display
560  if (!strStatusBarWarnings.isEmpty())
561  {
562  progressBarLabel->setText(strStatusBarWarnings);
563  progressBarLabel->setVisible(true);
564  progressBar->setVisible(false);
565  }
566 
567  QDateTime lastBlockDate = clientModel->getLastBlockDate();
568  int secs = lastBlockDate.secsTo(QDateTime::currentDateTime());
569  QString text;
570 
571  // Represent time from last generated block in human readable text
572  if(secs <= 0)
573  {
574  // Fully up to date. Leave text empty.
575  }
576  else if(secs < 60)
577  {
578  text = tr("%n second(s) ago","",secs);
