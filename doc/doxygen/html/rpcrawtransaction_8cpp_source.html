47 void TxToJSON(const CTransaction& tx, const uint256 hashBlock, Object& entry)
48 {
49  entry.push_back(Pair("txid", tx.GetHash().GetHex()));
50  entry.push_back(Pair("version", tx.nVersion));
51  entry.push_back(Pair("time", (int64_t)tx.nTime));
52  entry.push_back(Pair("locktime", (int64_t)tx.nLockTime));
53  entry.push_back(Pair("sendtoself", tx.IsSendToSelf()));
54  Array vin;
55  BOOST_FOREACH(const CTxIn& txin, tx.vin)
56  {
57  Object in;
58  if (tx.IsCoinBase())
59  in.push_back(Pair("coinbase", HexStr(txin.scriptSig.begin(), txin.scriptSig.end())));
60  else
61  {
62  in.push_back(Pair("txid", txin.prevout.hash.GetHex()));
63  in.push_back(Pair("vout", (int64_t)txin.prevout.n));
64  Object o;
65  o.push_back(Pair("asm", txin.scriptSig.ToString()));
66  o.push_back(Pair("hex", HexStr(txin.scriptSig.begin(), txin.scriptSig.end())));
67  in.push_back(Pair("scriptSig", o));
68  }
69  in.push_back(Pair("sequence", (int64_t)txin.nSequence));
70  vin.push_back(in);
71  }
72  entry.push_back(Pair("vin", vin));
73  Array vout;
74  for (unsigned int i = 0; i < tx.vout.size(); i++)
75  {
76  const CTxOut& txout = tx.vout[i];
77  Object out;
78  out.push_back(Pair("value", ValueFromAmount(txout.nValue)));
79  out.push_back(Pair("n", (int64_t)i));
80  Object o;
81  ScriptPubKeyToJSON(txout.scriptPubKey, o, false);
82  out.push_back(Pair("scriptPubKey", o));
83  vout.push_back(out);
84  }
85  entry.push_back(Pair("vout", vout));
86 
87  if (hashBlock != 0)
88  {
89  entry.push_back(Pair("blockhash", hashBlock.GetHex()));
90  map<uint256, CBlockIndex*>::iterator mi = mapBlockIndex.find(hashBlock);
91  if (mi != mapBlockIndex.end() && (*mi).second)
92  {
93  CBlockIndex* pindex = (*mi).second;
94  if (pindex->IsInMainChain())
95  {
96  entry.push_back(Pair("confirmations", 1 + nBestHeight - pindex->nHeight));
97  entry.push_back(Pair("time", (int64_t)pindex->nTime));
98  entry.push_back(Pair("blocktime", (int64_t)pindex->nTime));
99  }
100  else
101  entry.push_back(Pair("confirmations", 0));
102  }
103  }
104 }

646 Value signrawtransaction2(const Array& params, bool fHelp)
647 {
648  if (fHelp || params.size() < 1 || params.size() > 4)
649  throw runtime_error(
650  "signrawtransaction2 <hex string> [{\"redeemScript\":hex},...] [<privatekey1>,...] [sighashtype=\"ALL\"]\n"
651  "Sign inputs for raw transaction (serialized, hex-encoded).\n"
652  "Second optional argument (may be null) is an array of redeemScript.\n"
653  "Third optional argument (may be null) is an array of base58-encoded private\n"
654  "keys that, if given, will be the only keys used to sign the transaction.\n"
655  "Fourth optional argument is a string that is one of six values; ALL, NONE, SINGLE or\n"
656  "ALL|ANYONECANPAY, NONE|ANYONECANPAY, SINGLE|ANYONECANPAY.\n"
657  "Returns json object with keys:\n"
658  " hex : raw transaction with signature(s) (hex-encoded string)\n"
659  " complete : 1 if transaction has a complete set of signature (0 if not)"
660  + HelpRequiringPassphrase());
661 
662  RPCTypeCheck(params, list_of(str_type)(array_type)(array_type)(str_type), true);
663  vector<unsigned char> txData(ParseHex(params[0].get_str()));
664  CDataStream ssData(txData, SER_NETWORK, PROTOCOL_VERSION);
665  vector<CTransaction> txVariants;
666  while (!ssData.empty())
667  {
668  try {
669  CTransaction tx;
670  ssData >> tx;
671  txVariants.push_back(tx);
672  }
673  catch (std::exception &e) {
674  throw JSONRPCError(RPC_DESERIALIZATION_ERROR, "TX decode failed");
675  }
676  }
677 
678  if (txVariants.empty())
679  throw JSONRPCError(RPC_DESERIALIZATION_ERROR, "Missing transaction");
680 
681  // mergedTx will end up with all the signatures; it
682  // starts as a clone of the rawtx:
683  CTransaction mergedTx(txVariants[0]);
684  bool fComplete = true;
685 
686  // Fetch previous transactions (inputs):
687  map<COutPoint, CScript> mapPrevOut;
688  for (unsigned int i = 0; i < mergedTx.vin.size(); i++)
689  {
690  CTransaction tempTx;
691  MapPrevTx mapPrevTx;
692  CTxDB txdb("r");
693  map<uint256, CTxIndex> unused;
694  bool fInvalid;
695 
696  // FetchInputs aborts on failure, so we go one at a time.
697  tempTx.vin.push_back(mergedTx.vin[i]);
698  tempTx.FetchInputs(txdb, unused, false, false, mapPrevTx, fInvalid);
699 
700  // Copy results into mapPrevOut:
701  BOOST_FOREACH(const CTxIn& txin, tempTx.vin)
702  {
703  const uint256& prevHash = txin.prevout.hash;
704  if (mapPrevTx.count(prevHash) && mapPrevTx[prevHash].second.vout.size()>txin.prevout.n)
705  mapPrevOut[txin.prevout] = mapPrevTx[prevHash].second.vout[txin.prevout.n].scriptPubKey;
706  }
707  }
708 
709  bool fGivenKeys = false;
710  CBasicKeyStore tempKeystore;
711  if (params.size() > 2 && params[2].type() != null_type)
712  {
713  fGivenKeys = true;
714  Array keys = params[2].get_array();
715  BOOST_FOREACH(Value k, keys)
716  {
717  CBitcoinSecret vchSecret;
718  bool fGood = vchSecret.SetString(k.get_str());
719  if (!fGood)
720  throw JSONRPCError(RPC_INVALID_ADDRESS_OR_KEY, "Invalid private key");
721  CKey key;
722  bool fCompressed;
723  CSecret secret = vchSecret.GetSecret(fCompressed);
724  key.SetSecret(secret, fCompressed);
725  tempKeystore.AddKey(key);
726  }
727  }
728  else
729  EnsureWalletIsUnlocked();
730 
731  // Add previous txouts given in the RPC call:
732  if (params.size() > 1 && params[1].type() != null_type)
733  {
734  Array prevTxs = params[1].get_array();
735  BOOST_FOREACH(Value& p, prevTxs)
736  {
737  if (p.type() != obj_type)
738  throw JSONRPCError(RPC_DESERIALIZATION_ERROR, "expected object with {\"txid'\",\"vout\",\"scriptPubKey\"}");
739 
740  Object prevOut = p.get_obj();
741  // if redeemScript given and not using the local wallet (private keys
742  // given), add redeemScript to the tempKeystore so it can be signed:
743  if (fGivenKeys)
744  {
745  RPCTypeCheck(prevOut, map_list_of("redeemScript",str_type));
746  Value v = find_value(prevOut, "redeemScript");
747  if (!(v == Value::null))
748  {
749  vector<unsigned char> rsData(ParseHexV(v, "redeemScript"));
750  CScript redeemScript(rsData.begin(), rsData.end());
751  tempKeystore.AddCScript(redeemScript);
752  }
753  }
754  }
755  }
756  const CKeyStore& keystore = (fGivenKeys ? tempKeystore : *pwalletMain);
757  int nHashType = SIGHASH_ALL;
758  if (params.size() > 3 && params[3].type() != null_type)
759  {
760  static map<string, int> mapSigHashValues =
761  boost::assign::map_list_of
762  (string("ALL"), int(SIGHASH_ALL))
763  (string("ALL|ANYONECANPAY"), int(SIGHASH_ALL|SIGHASH_ANYONECANPAY))
764  (string("NONE"), int(SIGHASH_NONE))
765  (string("NONE|ANYONECANPAY"), int(SIGHASH_NONE|SIGHASH_ANYONECANPAY))
766  (string("SINGLE"), int(SIGHASH_SINGLE))
767  (string("SINGLE|ANYONECANPAY"), int(SIGHASH_SINGLE|SIGHASH_ANYONECANPAY))
768  ;
769  string strHashType = params[3].get_str();
770  if (mapSigHashValues.count(strHashType))
771  nHashType = mapSigHashValues[strHashType];
772  else
773  throw JSONRPCError(RPC_INVALID_PARAMETER, "Invalid sighash param");
774  }
775  bool fHashSingle = ((nHashType & ~SIGHASH_ANYONECANPAY) == SIGHASH_SINGLE);
776  // Sign what we can:
777  for (unsigned int i = 0; i < mergedTx.vin.size(); i++)
778  {
779  CTxIn& txin = mergedTx.vin[i];
780  if (mapPrevOut.count(txin.prevout) == 0)
781  {
782  fComplete = false;
783  continue;
784  }
785  const CScript& prevPubKey = mapPrevOut[txin.prevout];
786 
787  txin.scriptSig.clear();
788  // Only sign SIGHASH_SINGLE if there's a corresponding output:
789  if (!fHashSingle || (i < mergedTx.vout.size()))
790  SignSignature(keystore, prevPubKey, mergedTx, i, nHashType);
791 
792  // ... and merge in other signatures:
793  BOOST_FOREACH(const CTransaction& txv, txVariants)
794  {
795  txin.scriptSig = CombineSignatures(prevPubKey, mergedTx, i, txin.scriptSig, txv.vin[i].scriptSig);
796  }
797  if (!VerifyScript(txin.scriptSig, prevPubKey, mergedTx, i, 0))
798  fComplete = false;
799  }
800 
801  Object result;
802  CDataStream ssTx(SER_NETWORK, PROTOCOL_VERSION);
803  ssTx << mergedTx;
804  result.push_back(Pair("hex", HexStr(ssTx.begin(), ssTx.end())));
805  result.push_back(Pair("complete", fComplete));
806 
807  return result;
808 }
