49 // Perform ECDSA key recovery (see SEC1 4.1.6) for curves over (mod p)-fields
50 // recid selects which key is recovered
51 // if check is non-zero, additional checks are performed
52 int ECDSA_SIG_recover_key_GFp(EC_KEY *eckey, ECDSA_SIG *ecsig, const unsigned char *msg, int msglen, int recid, int check)
53 {
54  if (!eckey) return 0;
55 
56  int ret = 0;
57  BN_CTX *ctx = NULL;
58 
59  BIGNUM *x = NULL;
60  BIGNUM *e = NULL;
61  BIGNUM *order = NULL;
62  BIGNUM *sor = NULL;
63  BIGNUM *eor = NULL;
64  BIGNUM *field = NULL;
65  EC_POINT *R = NULL;
66  EC_POINT *O = NULL;
67  EC_POINT *Q = NULL;
68  BIGNUM *rr = NULL;
69  BIGNUM *zero = NULL;
70  int n = 0;
71  int i = recid / 2;
72 
73  const EC_GROUP *group = EC_KEY_get0_group(eckey);
74  if ((ctx = BN_CTX_new()) == NULL) { ret = -1; goto err; }
75  BN_CTX_start(ctx);
76  order = BN_CTX_get(ctx);
77  if (!EC_GROUP_get_order(group, order, ctx)) { ret = -2; goto err; }
78  x = BN_CTX_get(ctx);
79  if (!BN_copy(x, order)) { ret=-1; goto err; }
80  if (!BN_mul_word(x, i)) { ret=-1; goto err; }
81  if (!BN_add(x, x, ecsig->r)) { ret=-1; goto err; }
82  field = BN_CTX_get(ctx);
83  if (!EC_GROUP_get_curve_GFp(group, field, NULL, NULL, ctx)) { ret=-2; goto err; }
84  if (BN_cmp(x, field) >= 0) { ret=0; goto err; }
85  if ((R = EC_POINT_new(group)) == NULL) { ret = -2; goto err; }
86  if (!EC_POINT_set_compressed_coordinates_GFp(group, R, x, recid % 2, ctx)) { ret=0; goto err; }
87  if (check)
88  {
89  if ((O = EC_POINT_new(group)) == NULL) { ret = -2; goto err; }
90  if (!EC_POINT_mul(group, O, NULL, R, order, ctx)) { ret=-2; goto err; }
91  if (!EC_POINT_is_at_infinity(group, O)) { ret = 0; goto err; }
92  }
93  if ((Q = EC_POINT_new(group)) == NULL) { ret = -2; goto err; }
94  n = EC_GROUP_get_degree(group);
95  e = BN_CTX_get(ctx);
96  if (!BN_bin2bn(msg, msglen, e)) { ret=-1; goto err; }
97  if (8*msglen > n) BN_rshift(e, e, 8-(n & 7));
98  zero = BN_CTX_get(ctx);
99  if (!BN_zero(zero)) { ret=-1; goto err; }
100  if (!BN_mod_sub(e, zero, e, order, ctx)) { ret=-1; goto err; }
101  rr = BN_CTX_get(ctx);
102  if (!BN_mod_inverse(rr, ecsig->r, order, ctx)) { ret=-1; goto err; }
103  sor = BN_CTX_get(ctx);
104  if (!BN_mod_mul(sor, ecsig->s, rr, order, ctx)) { ret=-1; goto err; }
105  eor = BN_CTX_get(ctx);
106  if (!BN_mod_mul(eor, e, rr, order, ctx)) { ret=-1; goto err; }
107  if (!EC_POINT_mul(group, Q, eor, R, sor, ctx)) { ret=-2; goto err; }
108  if (!EC_KEY_set_public_key(eckey, Q)) { ret=-2; goto err; }
109 
110  ret = 1;
111 
112 err:
113  if (ctx) {
114  BN_CTX_end(ctx);
115  BN_CTX_free(ctx);
116  }
117  if (R != NULL) EC_POINT_free(R);
118  if (O != NULL) EC_POINT_free(O);
119  if (Q != NULL) EC_POINT_free(Q);
120  return ret;
121 }

177 int CompareBigEndian(const unsigned char *c1, size_t c1len, const unsigned char *c2, size_t c2len) {
178  while (c1len > c2len) {
179  if (*c1)
180  return 1;
181  c1++;
182  c1len--;
183  }
184  while (c2len > c1len) {
185  if (*c2)
186  return -1;
187  c2++;
188  c2len--;
189  }
190  while (c1len > 0) {
191  if (*c1 > *c2)
192  return 1;
193  if (*c2 > *c1)
194  return -1;
195  c1++;
196  c2++;
197  c1len--;
198  }
199  return 0;
200 }

334 bool CKey::Sign(uint256 hash, std::vector<unsigned char>& vchSig)
335 {
336  vchSig.clear();
337  ECDSA_SIG *sig = ECDSA_do_sign((unsigned char*)&hash, sizeof(hash), pkey);
338  if (sig == NULL)
339  return false;
340  BN_CTX *ctx = BN_CTX_new();
341  BN_CTX_start(ctx);
342  const EC_GROUP *group = EC_KEY_get0_group(pkey);
343  BIGNUM *order = BN_CTX_get(ctx);
344  BIGNUM *halforder = BN_CTX_get(ctx);
345  EC_GROUP_get_order(group, order, ctx);
346  BN_rshift1(halforder, order);
347  if (BN_cmp(sig->s, halforder) > 0) {
348  // enforce low S values, by negating the value (modulo the order) if above order/2.
349  BN_sub(sig->s, order, sig->s);
350  }
351  BN_CTX_end(ctx);
352  BN_CTX_free(ctx);
353  unsigned int nSize = ECDSA_size(pkey);
354  vchSig.resize(nSize); // Make sure it is big enough
355  unsigned char *pos = &vchSig[0];
356  nSize = i2d_ECDSA_SIG(sig, &pos);
357  ECDSA_SIG_free(sig);
358  vchSig.resize(nSize); // Shrink to fit actual size
359  return true;
360 }

362 // create a compact signature (65 bytes), which allows reconstructing the used public key
363 // The format is one header byte, followed by two times 32 bytes for the serialized r and s values.
364 // The header byte: 0x1B = first key with even y, 0x1C = first key with odd y,
365 // 0x1D = second key with even y, 0x1E = second key with odd y
366 bool CKey::SignCompact(uint256 hash, std::vector<unsigned char>& vchSig)
367 {
368  bool fOk = false;
369  ECDSA_SIG *sig = ECDSA_do_sign((unsigned char*)&hash, sizeof(hash), pkey);
370  if (sig==NULL)
371  return false;
372  vchSig.clear();
373  vchSig.resize(65,0);
374  int nBitsR = BN_num_bits(sig->r);
375  int nBitsS = BN_num_bits(sig->s);
376  if (nBitsR <= 256 && nBitsS <= 256)
377  {
378  int nRecId = -1;
379  for (int i=0; i<4; i++)
380  {
381  CKey keyRec;
382  keyRec.fSet = true;
383  if (fCompressedPubKey)
384  keyRec.SetCompressedPubKey();
385  if (ECDSA_SIG_recover_key_GFp(keyRec.pkey, sig, (unsigned char*)&hash, sizeof(hash), i, 1) == 1)
386  if (keyRec.GetPubKey() == this->GetPubKey())
387  {
388  nRecId = i;
389  break;
390  }
391  }
392 
393  if (nRecId == -1)
394  {
395  ECDSA_SIG_free(sig);
396  throw key_error("CKey::SignCompact() : unable to construct recoverable key");
397  }
398 
399  vchSig[0] = nRecId+27+(fCompressedPubKey ? 4 : 0);
400  BN_bn2bin(sig->r,&vchSig[33-(nBitsR+7)/8]);
401  BN_bn2bin(sig->s,&vchSig[65-(nBitsS+7)/8]);
402  fOk = true;
403  }
404  ECDSA_SIG_free(sig);
405  return fOk;
406 }

410 // If this function succeeds, the recovered public key is guaranteed to be valid
