672  if (nDepth >= 1)

682  std::vector<const CMerkleTx*> vWorkQueue;
683  vWorkQueue.reserve(vtxPrev.size()+1);
684  vWorkQueue.push_back(this);
685  for (unsigned int i = 0; i < vWorkQueue.size(); i++)
686  {

688

690  return false;
691  int nPDepth = ptx->GetDepthInMainChain();
692  if (nPDepth >= 1)
693  continue;
694  if (nPDepth < 0)
695  return false;
696  if (!pwallet->IsFromMe(*ptx))
697  return false;
698 
699  if (mapPrev.empty())
700  {
701  BOOST_FOREACH(const CMerkleTx& tx, vtxPrev)
702  mapPrev[tx.GetHash()] = &tx;
703  }
704 
705  BOOST_FOREACH(const CTxIn& txin, ptx->vin)
706  {
707  if (!mapPrev.count(txin.prevout.hash))
708  return false;
709  vWorkQueue.push_back(mapPrev[txin.prevout.hash]);
710  }
711  }
712 
713  return true;
