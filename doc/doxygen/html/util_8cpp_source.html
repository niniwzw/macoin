393 bool ParseMoney(const char* pszIn, int64_t& nRet)
394 {
395  string strWhole;
396  int64_t nUnits = 0;
397  const char* p = pszIn;
398  while (isspace(*p))
399  p++;
400  for (; *p; p++)
401  {
402  if (*p == '.')
403  {
404  p++;
405  int64_t nMult = CENT*10;
406  while (isdigit(*p) && (nMult > 0))
407  {
408  nUnits += nMult * (*p++ - '0');
409  nMult /= 10;
410  }
411  break;
412  }
413  if (isspace(*p))
414  break;
415  if (!isdigit(*p))
416  return false;
417  strWhole.insert(strWhole.end(), *p);
418  }
419  for (; *p; p++)
420  if (!isspace(*p))
421  return false;
422  if (strWhole.size() > 10) // guard against 63 bit overflow
423  return false;
424  if (nUnits < 0 || nUnits > COIN)
425  return false;
426  int64_t nWhole = atoi64(strWhole);
427  int64_t nValue = nWhole*COIN + nUnits;
428

594 string EncodeBase64(const unsigned char* pch, size_t len)
595 {
596  static const char *pbase64 = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/";
597 
598  string strRet="";
599  strRet.reserve((len+2)/3*4);
600 
601  int mode=0, left=0;
602  const unsigned char *pchEnd = pch+len;
603 
604  while (pch<pchEnd)
605  {
606  int enc = *(pch++);
607  switch (mode)
608  {
609  case 0: // we have no bits
610  strRet += pbase64[enc >> 2];
611  left = (enc & 3) << 4;
612  mode = 1;
613  break;
614 
615  case 1: // we have two bits
616  strRet += pbase64[left | (enc >> 4)];
617  left = (enc & 15) << 2;
618  mode = 2;
619  break;
620 
621  case 2: // we have four bits
622  strRet += pbase64[left | (enc >> 6)];
623  strRet += pbase64[enc & 63];
624  mode = 0;
625  break;
626  }
627  }
628 
629  if (mode)
630  {
631  strRet += pbase64[left];
632  strRet += '=';
633  if (mode == 1)
634  strRet += '=';
635  }
636 
637  return strRet;
638 }
