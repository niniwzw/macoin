159 //
160 // Compact size
161 // size < 253 -- 1 byte
162 // size <= USHRT_MAX -- 3 bytes (253 + 2 bytes)
163 // size <= UINT_MAX -- 5 bytes (254 + 4 bytes)
164 // size > UINT_MAX -- 9 bytes (255 + 8 bytes)
165 //
166 inline unsigned int GetSizeOfCompactSize(uint64_t nSize)
167 {
168  if (nSize < 253) return sizeof(unsigned char);
169  else if (nSize <= std::numeric_limits<unsigned short>::max()) return sizeof(unsigned char) + sizeof(unsigned short);
170  else if (nSize <= std::numeric_limits<unsigned int>::max()) return sizeof(unsigned char) + sizeof(unsigned int);
171  else return sizeof(unsigned char) + sizeof(uint64_t);
172 }

174 template<typename Stream>
175 void WriteCompactSize(Stream& os, uint64_t nSize)
176 {
177  if (nSize < 253)
178  {
179  unsigned char chSize = nSize;
180  WRITEDATA(os, chSize);
181  }
182  else if (nSize <= std::numeric_limits<unsigned short>::max())
183  {
184  unsigned char chSize = 253;
185  unsigned short xSize = nSize;
186  WRITEDATA(os, chSize);
187  WRITEDATA(os, xSize);
188  }
189  else if (nSize <= std::numeric_limits<unsigned int>::max())
190  {
191  unsigned char chSize = 254;
192  unsigned int xSize = nSize;
193  WRITEDATA(os, chSize);
194  WRITEDATA(os, xSize);
195  }
196  else
197  {
198  unsigned char chSize = 255;
199  uint64_t xSize = nSize;
200  WRITEDATA(os, chSize);
201  WRITEDATA(os, xSize);
202  }
203  return;
204 }

206 template<typename Stream>
207 uint64_t ReadCompactSize(Stream& is)
208 {
209  unsigned char chSize;
210  READDATA(is, chSize);
211  uint64_t nSizeRet = 0;
212  if (chSize < 253)
213  {
214  nSizeRet = chSize;
215  }
216  else if (chSize == 253)
217  {
218  unsigned short xSize;
219  READDATA(is, xSize);
220  nSizeRet = xSize;
221  if (nSizeRet < 253)
222  throw std::ios_base::failure("non-canonical ReadCompactSize()");
223  }
224  else if (chSize == 254)
225  {
226  unsigned int xSize;
227  READDATA(is, xSize);
228  nSizeRet = xSize;
229  if (nSizeRet < 0x10000u)
230  throw std::ios_base::failure("non-canonical ReadCompactSize()");
231  }
232  else
233  {
234  uint64_t xSize;
235  READDATA(is, xSize);
236  nSizeRet = xSize;
237  if (nSizeRet < 0x100000000LLu)
238  throw std::ios_base::failure("non-canonical ReadCompactSize()");
239  }
240  if (nSizeRet > (uint64_t)MAX_SIZE)
241  throw std::ios_base::failure("ReadCompactSize() : size too large");
242  return nSizeRet;
243 }

439 template<typename Stream, typename T, typename A>
440 void Unserialize_impl(Stream& is, std::vector<T, A>& v, int nType, int nVersion, const boost::true_type&)
441 {
442  // Limit size per read so bogus size value won't cause out of memory
443  v.clear();
444  unsigned int nSize = ReadCompactSize(is);
445  unsigned int i = 0;
446  while (i < nSize)
447  {
448  unsigned int blk = std::min(nSize - i, (unsigned int)(1 + 4999999 / sizeof(T)));
449  v.resize(i + blk);
450  is.read((char*)&v[i], blk * sizeof(T));
451  i += blk;
452  }
453 }
