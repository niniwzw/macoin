165  void setint64(int64_t sn)
166  {
167  unsigned char pch[sizeof(sn) + 6];
168  unsigned char* p = pch + 4;
169  bool fNegative;
170  uint64_t n;
171 
172  if (sn < (int64_t)0)
173  {
174  // Since the minimum signed integer cannot be represented as positive so long as its type is signed, and it's not well-defined what happens if you make it unsigned before negating it, we instead increment the negative integer by 1, convert it, then increment the (now positive) unsigned integer by 1 to compensate
175  n = -(sn + 1);
176  ++n;
177  fNegative = true;
178  } else {
179  n = sn;
180  fNegative = false;
181  }
182 
183  bool fLeadingZeroes = true;
184  for (int i = 0; i < 8; i++)
185  {
186  unsigned char c = (n >> 56) & 0xff;
187  n <<= 8;
188  if (fLeadingZeroes)
189  {
190  if (c == 0)
191  continue;
192  if (c & 0x80)
193  *p++ = (fNegative ? 0x80 : 0);
194  else if (fNegative)
195  c |= 0x80;
196  fLeadingZeroes = false;
197  }
198  *p++ = c;
199  }
200  unsigned int nSize = p - (pch + 4);
201  pch[0] = (nSize >> 24) & 0xff;
202  pch[1] = (nSize >> 16) & 0xff;
203  pch[2] = (nSize >> 8) & 0xff;
204  pch[3] = (nSize) & 0xff;
205  BN_mpi2bn(pch, p - pch, this);
206  }

295  {
296  std::vector<unsigned char> vch2(vch.size() + 4);
