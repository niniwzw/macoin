37 void SplitHostPort(std::string in, int &portOut, std::string &hostOut) {
38  size_t colon = in.find_last_of(':');
39  // if a : is found, and it either follows a [...], or no other : is in the string, treat it as port separator
40  bool fHaveColon = colon != in.npos;
41  bool fBracketed = fHaveColon && (in[0]=='[' && in[colon-1]==']'); // if there is a colon, and in[0]=='[', colon is not 0, so in[colon-1] is safe
42  bool fMultiColon = fHaveColon && (in.find_last_of(':',colon-1) != in.npos);
43  if (fHaveColon && (colon==0 || fBracketed || !fMultiColon)) {
44  char *endp = NULL;
45  int n = strtol(in.c_str() + colon + 1, &endp, 10);
46  if (endp && *endp == 0 && n >= 0) {
47  in = in.substr(0, colon);
48  if (n > 0 && n < 0x10000)
49  portOut = n;
50  }
51  }
52  if (in.size()>0 && in[0] == '[' && in[in.size()-1] == ']')
53  hostOut = in.substr(1, in.size()-2);
54  else
55  hostOut = in;
56 }

185  }

193  {
194  closesocket(hSocket);

196  printf("ERROR: Proxy returned error %d\n", pchRet[1]);
197  return false;

204 {
205  printf("SOCKS5 connecting %s\n", strDest.c_str());
206  if (strDest.size() > 255)

217  {
218  closesocket(hSocket);

220  }
221  char pchRet1[2];
222  if (recv(hSocket, pchRet1, 2, 0) != 2)
223  {
224  closesocket(hSocket);

226  }

228  {
229  closesocket(hSocket);

231  }

234  strSocks5 += static_cast<char>(std::min((int)strDest.size(), 255));

237  strSocks5 += static_cast<char>((port >> 0) & 0xFF);
238  ret = send(hSocket, strSocks5.c_str(), strSocks5.size(), MSG_NOSIGNAL);
239  if (ret != (ssize_t)strSocks5.size())
240  {
241  closesocket(hSocket);
242  return error("Error sending to proxy");
243  }
244  char pchRet2[4];
245  if (recv(hSocket, pchRet2, 4, 0) != 4)
246  {
247  closesocket(hSocket);
248  return error("Error reading proxy response");
249  }
250  if (pchRet2[0] != 0x05)
251  {
252  closesocket(hSocket);
253  return error("Proxy failed to accept request");
254  }

259  {

261  case 0x02: return error("Proxy error: connection not allowed");
262  case 0x03: return error("Proxy error: network unreachable");
263  case 0x04: return error("Proxy error: host unreachable");
264  case 0x05: return error("Proxy error: connection refused");
265  case 0x06: return error("Proxy error: TTL expired");
266  case 0x07: return error("Proxy error: protocol error");
267  case 0x08: return error("Proxy error: address type not supported");
268  default: return error("Proxy error: unknown");
269  }
270  }
271  if (pchRet2[2] != 0x00)
272  {
273  closesocket(hSocket);
274  return error("Error: malformed proxy response");
275  }
276  char pchRet3[256];
277  switch (pchRet2[3])
278  {
279  case 0x01: ret = recv(hSocket, pchRet3, 4, 0) != 4; break;
280  case 0x04: ret = recv(hSocket, pchRet3, 16, 0) != 16; break;
281  case 0x03:
282  {
283  ret = recv(hSocket, pchRet3, 1, 0) != 1;
284  if (ret) {
285  closesocket(hSocket);
286  return error("Error reading from proxy");
287  }
288  int nRecv = pchRet3[0];
289  ret = recv(hSocket, pchRet3, nRecv, 0) != nRecv;
290  break;
291  }
292  default: closesocket(hSocket); return error("Error: malformed proxy response");
293  }
294  if (ret)
295  {
296  closesocket(hSocket);
297  return error("Error reading from proxy");
298  }
299  if (recv(hSocket, pchRet3, 2, 0) != 2)
300  {
301  closesocket(hSocket);
302  return error("Error reading from proxy");
303  }
304  printf("SOCKS5 connected %s\n", strDest.c_str());
305  return true;

308 bool static ConnectSocketDirectly(const CService &addrConnect, SOCKET& hSocketRet, int nTimeout)
309 {
310  hSocketRet = INVALID_SOCKET;
311 
312  struct sockaddr_storage sockaddr;
313  socklen_t len = sizeof(sockaddr);
314  if (!addrConnect.GetSockAddr((struct sockaddr*)&sockaddr, &len)) {
315  printf("Cannot connect to %s: unsupported network\n", addrConnect.ToString().c_str());
316  return false;
317  }
318 
319  SOCKET hSocket = socket(((struct sockaddr*)&sockaddr)->sa_family, SOCK_STREAM, IPPROTO_TCP);
320  if (hSocket == INVALID_SOCKET)
321  return false;
322 #ifdef SO_NOSIGPIPE
323  int set = 1;
324  setsockopt(hSocket, SOL_SOCKET, SO_NOSIGPIPE, (void*)&set, sizeof(int));
325 #endif
326 
327 #ifdef WIN32
328  u_long fNonblock = 1;
329  if (ioctlsocket(hSocket, FIONBIO, &fNonblock) == SOCKET_ERROR)
330 #else
331  int fFlags = fcntl(hSocket, F_GETFL, 0);
332  if (fcntl(hSocket, F_SETFL, fFlags | O_NONBLOCK) == -1)
333 #endif
334  {
335  closesocket(hSocket);
336  return false;
337  }
338 
339  if (connect(hSocket, (struct sockaddr*)&sockaddr, len) == SOCKET_ERROR)
340  {
341  int nErr = WSAGetLastError();
342  // WSAEINVAL is here because some legacy version of winsock uses it
343  if (nErr == WSAEINPROGRESS || nErr == WSAEWOULDBLOCK || nErr == WSAEINVAL)
344  {
345  struct timeval timeout;
346  timeout.tv_sec = nTimeout / 1000;
347  timeout.tv_usec = (nTimeout % 1000) * 1000;
348 
349  fd_set fdset;
350  FD_ZERO(&fdset);
351  FD_SET(hSocket, &fdset);
352  int nRet = select(hSocket + 1, NULL, &fdset, NULL, &timeout);
353  if (nRet == 0)
354  {
355  printf("connection timeout\n");
356  closesocket(hSocket);
357  return false;
358  }
359  if (nRet == SOCKET_ERROR)
360  {
361  printf("select() for connection failed: %i\n",WSAGetLastError());
362  closesocket(hSocket);
363  return false;
364  }
365  socklen_t nRetSize = sizeof(nRet);
366 #ifdef WIN32
367  if (getsockopt(hSocket, SOL_SOCKET, SO_ERROR, (char*)(&nRet), &nRetSize) == SOCKET_ERROR)
368 #else
369  if (getsockopt(hSocket, SOL_SOCKET, SO_ERROR, &nRet, &nRetSize) == SOCKET_ERROR)
370 #endif
371  {
372  printf("getsockopt() for connection failed: %i\n",WSAGetLastError());
373  closesocket(hSocket);
374  return false;
375  }
376  if (nRet != 0)
377  {
378  printf("connect() failed after select(): %s\n",strerror(nRet));
379  closesocket(hSocket);
380  return false;
381  }
382  }
383 #ifdef WIN32
384  else if (WSAGetLastError() != WSAEISCONN)
385 #else
386  else
387 #endif
388  {
389  printf("connect() failed: %i\n",WSAGetLastError());
390  closesocket(hSocket);
391  return false;
392  }
393  }
394 
395  // this isn't even strictly necessary
396  // CNode::ConnectNode immediately turns the socket back to non-blocking
397  // but we'll turn it back to blocking just in case
398 #ifdef WIN32
399  fNonblock = 0;
400  if (ioctlsocket(hSocket, FIONBIO, &fNonblock) == SOCKET_ERROR)
401 #else
402  fFlags = fcntl(hSocket, F_GETFL, 0);
403  if (fcntl(hSocket, F_SETFL, fFlags & !O_NONBLOCK) == SOCKET_ERROR)
404 #endif
405  {
406  closesocket(hSocket);
407  return false;
408  }
409 
410  hSocketRet = hSocket;
411  return true;
412 }

705 bool CNetAddr::IsValid() const
706 {

708  // of addr messages from versions before 0.2.9 checksum.
709  // Two consecutive addr messages look like this:
710  // header20 vectorlen3 addr26 addr26 addr26 header20 vectorlen3 addr26 addr26 addr26...

712  // of addr misaligned by 3 bytes.

714  return false;

718  if (memcmp(ip, ipNone, 16) == 0)
719  return false;

723  return false;

728  uint32_t ipNone = INADDR_NONE;
729  if (memcmp(ip+12, &ipNone, 4) == 0)
730  return false;

732  // 0
733  ipNone = 0;
734  if (memcmp(ip+12, &ipNone, 4) == 0)

736  }

821 // get canonical identifier of an address' group
822 // no two connections will be attempted to addresses with the same group
823 std::vector<unsigned char> CNetAddr::GetGroup() const
824 {
825  std::vector<unsigned char> vchRet;
826  int nClass = NET_IPV6;
827  int nStartByte = 0;
828  int nBits = 16;
829 
830  // all local addresses belong to the same group
831  if (IsLocal())
832  {
833  nClass = 255;
834  nBits = 0;
835  }
836 
837  // all unroutable addresses belong to the same group
838  if (!IsRoutable())
839  {
840  nClass = NET_UNROUTABLE;
841  nBits = 0;
842  }
843  // for IPv4 addresses, '1' + the 16 higher-order bits of the IP
844  // includes mapped IPv4, SIIT translated IPv4, and the well-known prefix
845  else if (IsIPv4() || IsRFC6145() || IsRFC6052())
846  {
847  nClass = NET_IPV4;
848  nStartByte = 12;
849  }
850  // for 6to4 tunnelled addresses, use the encapsulated IPv4 address
851  else if (IsRFC3964())
852  {
853  nClass = NET_IPV4;
854  nStartByte = 2;
855  }
856  // for Teredo-tunnelled IPv6 addresses, use the encapsulated IPv4 address
857  else if (IsRFC4380())
858  {
859  vchRet.push_back(NET_IPV4);
860  vchRet.push_back(GetByte(3) ^ 0xFF);
861  vchRet.push_back(GetByte(2) ^ 0xFF);
862  return vchRet;
863  }
864  else if (IsTor())
865  {
866  nClass = NET_TOR;
867  nStartByte = 6;
868  nBits = 4;
869  }
870  else if (IsI2P())
871  {
872  nClass = NET_I2P;
873  nStartByte = 6;
874  nBits = 4;
875  }
876  // for he.net, use /36 groups
877  else if (GetByte(15) == 0x20 && GetByte(14) == 0x11 && GetByte(13) == 0x04 && GetByte(12) == 0x70)
878  nBits = 36;
879  // for the rest of the IPv6 network, use /32 groups
880  else
881  nBits = 32;
882 
883  vchRet.push_back(nClass);
884  while (nBits >= 8)
885  {
886  vchRet.push_back(GetByte(15 - nStartByte));
887  nStartByte++;
888  nBits -= 8;
889  }
890  if (nBits > 0)
891  vchRet.push_back(GetByte(15 - nStartByte) | ((1 << nBits) - 1));
892 
893  return vchRet;
894 }

923 int CNetAddr::GetReachabilityFrom(const CNetAddr *paddrPartner) const
924 {
925  enum Reachability {
926  REACH_UNREACHABLE,
927  REACH_DEFAULT,
928  REACH_TEREDO,
929  REACH_IPV6_WEAK,
930  REACH_IPV4,
931  REACH_IPV6_STRONG,
932  REACH_PRIVATE
933  };
934 
935  if (!IsRoutable())
936  return REACH_UNREACHABLE;
937 
938  int ourNet = GetExtNetwork(this);
939  int theirNet = GetExtNetwork(paddrPartner);
940  bool fTunnel = IsRFC3964() || IsRFC6052() || IsRFC6145();
941 
942  switch(theirNet) {
943  case NET_IPV4:
944  switch(ourNet) {
945  default: return REACH_DEFAULT;
946  case NET_IPV4: return REACH_IPV4;
947  }
948  case NET_IPV6:
949  switch(ourNet) {
950  default: return REACH_DEFAULT;
951  case NET_TEREDO: return REACH_TEREDO;
952  case NET_IPV4: return REACH_IPV4;
953  case NET_IPV6: return fTunnel ? REACH_IPV6_WEAK : REACH_IPV6_STRONG; // only prefer giving our IPv6 address if it's not tunnelled
954  }
955  case NET_TOR:
956  switch(ourNet) {
957  default: return REACH_DEFAULT;
958  case NET_IPV4: return REACH_IPV4; // Tor users can connect to IPv4 as well
959  case NET_TOR: return REACH_PRIVATE;
960  }
961  case NET_I2P:
962  switch(ourNet) {
963  default: return REACH_DEFAULT;
964  case NET_I2P: return REACH_PRIVATE;
965  }
966  case NET_TEREDO:
967  switch(ourNet) {
968  default: return REACH_DEFAULT;
969  case NET_TEREDO: return REACH_TEREDO;
970  case NET_IPV6: return REACH_IPV6_WEAK;
971  case NET_IPV4: return REACH_IPV4;
972  }
973  case NET_UNKNOWN:
974  case NET_UNROUTABLE:
975  default:
976  switch(ourNet) {
977  default: return REACH_DEFAULT;
978  case NET_TEREDO: return REACH_TEREDO;
979  case NET_IPV6: return REACH_IPV6_WEAK;
980  case NET_IPV4: return REACH_IPV4;
981  case NET_I2P: return REACH_PRIVATE; // assume connections from unroutable addresses are
982  case NET_TOR: return REACH_PRIVATE; // either from Tor/I2P, or don't care about our address
983  }
984  }
985 }
