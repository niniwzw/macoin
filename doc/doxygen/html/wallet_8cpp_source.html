1203  vector<char>& vfBest, int64_t& nBest, int iterations = 1000)
1204 {

1206 
1207  vfBest.assign(vValue.size(), true);
1208  nBest = nTotalLower;
1209 
1210  seed_insecure_rand();
1211

1213  {
1214  vfIncluded.assign(vValue.size(), false);
1215  int64_t nTotal = 0;
1216  bool fReachedTarget = false;
1217  for (int nPass = 0; nPass < 2 && !fReachedTarget; nPass++)
1218  {
1219  for (unsigned int i = 0; i < vValue.size(); i++)
1220  {

1225  //because there may be some privacy improvement by making
1226  //the selection random.
1227  if (nPass == 0 ? insecure_rand()&1 : !vfIncluded[i])

1230  vfIncluded[i] = true;

1232  {
1233  fReachedTarget = true;
1234  if (nTotal < nBest)
1235  {
1236  nBest = nTotal;
1237  vfBest = vfIncluded;
1238  }
