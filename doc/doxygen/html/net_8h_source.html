397  void AskFor(const CInv& inv)
398  {
399  // We're using mapAskFor as a priority queue,
400  // the key is the earliest time the request can be sent
401  int64_t& nRequestTime = mapAlreadyAskedFor[inv];
402  if (fDebugNet)
403  printf("askfor %s %"PRId64" (%s)\n", inv.ToString().c_str(), nRequestTime, DateTimeStrFormat("%H:%M:%S", nRequestTime/1000000).c_str());
404 
405  // Make sure not to reuse time indexes to keep things in the same order
406  int64_t nNow = (GetTime() - 1) * 1000000;
407  static int64_t nLastTime;
408  ++nLastTime;
409  nNow = std::max(nNow, nLastTime);
410  nLastTime = nNow;
411 
412  // Each retry is 2 minutes after the last
413  nRequestTime = std::max(nRequestTime + 2 * 60 * 1000000, nNow);
414  mapAskFor.insert(std::make_pair(nRequestTime, inv));
415  }

438  void EndMessage()
439  {
440  if (mapArgs.count("-dropmessagestest") && GetRand(atoi(mapArgs["-dropmessagestest"])) == 0)
441  {
442  printf("dropmessages DROPPING SEND MESSAGE\n");
443  AbortMessage();
444  return;
445  }
446 
447  if (ssSend.size() == 0)
448  return;
449 
450  // Set the size
451  unsigned int nSize = ssSend.size() - CMessageHeader::HEADER_SIZE;
452  memcpy((char*)&ssSend[CMessageHeader::MESSAGE_SIZE_OFFSET], &nSize, sizeof(nSize));
453 
454  // Set the checksum
455  uint256 hash = Hash(ssSend.begin() + CMessageHeader::HEADER_SIZE, ssSend.end());
456  unsigned int nChecksum = 0;
457  memcpy(&nChecksum, &hash, sizeof(nChecksum));
458  assert(ssSend.size () >= CMessageHeader::CHECKSUM_OFFSET + sizeof(nChecksum));
459  memcpy((char*)&ssSend[CMessageHeader::CHECKSUM_OFFSET], &nChecksum, sizeof(nChecksum));
460 
461  if (fDebug) {
462  printf("(%d bytes)\n", nSize);
463  }
464 
465  std::deque<CSerializeData>::iterator it = vSendMsg.insert(vSendMsg.end(), CSerializeData());
466  ssSend.GetAndClear(*it);
467  nSendSize += (*it).size();
468 
469  // If write queue empty, attempt "optimistic write"
470  if (it == vSendMsg.begin())
471  SocketSendData(this);
472 
473  LEAVE_CRITICAL_SECTION(cs_vSend);
474  }
