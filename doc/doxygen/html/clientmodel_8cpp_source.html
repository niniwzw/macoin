61 void ClientModel::updateTimer()
62 {
63  // Get required lock upfront. This avoids the GUI from getting stuck on
64  // periodical polls if the core is holding the locks for a longer time -
65  // for example, during a wallet rescan.
66  TRY_LOCK(cs_main, lockMain);
67  if(!lockMain)
68  return;
69  // Some quantities (such as number of blocks) change so fast that we don't want to be notified for each change.
70  // Periodically check and update with a timer.
71  int newNumBlocks = getNumBlocks();
72  int newNumBlocksOfPeers = getNumBlocksOfPeers();
73 
74  if(cachedNumBlocks != newNumBlocks || cachedNumBlocksOfPeers != newNumBlocksOfPeers)
75  {
76  cachedNumBlocks = newNumBlocks;
77  cachedNumBlocksOfPeers = newNumBlocksOfPeers;
78
